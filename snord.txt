alert icmp any any -> $HOME_NET any (msg:"ICMP Large Packet >100 bytes"; dsize:>100; sid:100001; rev:1;)
alert icmp any any -> $HOME_NET any (msg:"ICMP Flood Detected"; detection_filter:track by_src, count 10, seconds 10; sid:100002; rev:1;)

alert tcp any any -> $HOME_NET 80 (msg:"SYN Flood HTTP"; flags:S; detection_filter:track by_src, count 10, seconds 10; sid:100003; rev:1;)
alert tcp any any -> $HOME_NET 22 (msg:"SSH Brute Force"; flags:S; detection_filter:track by_src, count 5, seconds 10; sid:100004; rev:1;)

alert udp any any -> any 53 (msg:"DNS Gmail Access"; content:"gmail"; sid:100005; rev:1;)
alert udp any any -> any 53 (msg:"DNS Facebook Access"; content:"facebook"; sid:100006; rev:1;)
alert udp any any -> any 53 (msg:"DNS Instagram Access"; content:"instagram"; sid:100007; rev:1;)
alert tcp any any -> $HOME_NET any (msg:"Port Scan Detected"; flags:S; detection_filter:track by_src, count 15, seconds 10; sid:100008; rev:1;)
alert tcp any any -> $HOME_NET 80 (msg:"HTTP GET Flood"; content:"GET"; http_method; detection_filter:track by_src, count 20, seconds 10; sid:100009; rev:1;)
alert icmp any any -> $HOME_NET any (msg:"ICMP Activity Spike"; detection_filter:track by_src, count 20, seconds 10; sid:100010; rev:1;)
alert tcp any any -> $HOME_NET 22 (msg:"SSH Connection Attempt"; flags:S; detection_filter:track by_src, count 5, seconds 10; sid:100011; rev:1;)
